from sklearn.decomposition import PCA
import matplotlib.pyplot as plt

# Crear volumen y preparar datos
df['volumen_cm3'] = df['ancho_cm'] * df['largo_cm'] * df['alto_cm']
clustering_data = df[['volumen_cm3', 'peso_kg', 'manipulacion', 'temperatura']].copy()
clustering_data = pd.get_dummies(clustering_data, columns=['manipulacion', 'temperatura'], drop_first=True)

# Escalar y agrupar
scaler = StandardScaler()
X_scaled = scaler.fit_transform(clustering_data)
kmeans = KMeans(n_clusters=4, random_state=42, n_init=10)
clusters = kmeans.fit_predict(X_scaled)

# PCA para graficar
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_scaled)

# Graficar clusters
plt.figure(figsize=(10, 6))
scatter = plt.scatter(X_pca[:, 0], X_pca[:, 1], c=clusters, cmap='tab10', alpha=0.6)
plt.title("Agrupación de Productos en 4 Depósitos (KMeans + PCA)")
plt.xlabel("Componente Principal 1")
plt.ylabel("Componente Principal 2")
plt.colorbar(scatter, label="Clúster/Depósito")
plt.grid(True)
plt.tight_layout()
plt.show()
